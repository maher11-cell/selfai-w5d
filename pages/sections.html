Developer API

Six SDKs. One shape.

// Run a workflow
import { W5D } from "@w5d/sdk";
const w5d = new W5D({ apiKey: process.env.W5D_KEY });
const run = await w5d.workflows.run("qbr-recap", {
  account: "northwind",
});
console.log(run.output);
from w5d import Client
client = Client(api_key=os.environ["W5D_KEY"])
run = client.workflows.run("qbr-recap", account="northwind")
print(run.output)
c := w5d.New(os.Getenv("W5D_KEY"))
run, _ := c.Workflows.Run("qbr-recap", w5d.M{"account": "northwind"})
fmt.Println(run.Output)
curl https://api.w5d.ai/v1/workflows/qbr-recap/run \
  -H "Authorization: Bearer $W5D_KEY" \
  -d '{"account":"northwind"}'
